#### 步骤4: 实现上下文菜单逻辑

在 `player.js` 中添加了完整的上下文菜单处理逻辑：

**菜单显示函数**：
```javascript
function showContextMenu(e, index, item) {
    contextMenuTargetItem = item;
    contextMenuTargetIndex = index;

    contextMenu.style.left = e.clientX + 'px';
    contextMenu.style.top = e.clientY + 'px';
    contextMenu.classList.add('show');

    // 边界检查，防止菜单超出窗口
    const menuWidth = contextMenu.offsetWidth;
    const menuHeight = contextMenu.offsetHeight;
    const windowWidth = window.innerWidth;
    const windowHeight = window.innerHeight;

    if (e.clientX + menuWidth > windowWidth) {
        contextMenu.style.left = (windowWidth - menuWidth - 5) + 'px';
    }
    if (e.clientY + menuHeight > windowHeight) {
        contextMenu.style.top = (windowHeight - menuHeight - 5) + 'px';
    }
}
```

**菜单隐藏函数**：
```javascript
function hideContextMenu() {
    contextMenu.classList.remove('show');
    contextMenuTargetItem = null;
    contextMenuTargetIndex = null;
}
```

**菜单点击处理**：
```javascript
contextMenu.addEventListener('click', async (e) => {
    e.stopPropagation();
    const menuItem = e.target.closest('.context-menu-item');
    if (!menuItem || !contextMenuTargetItem) return;

    const action = menuItem.dataset.action;

    switch (action) {
        case 'play':
            playItem(contextMenuTargetIndex);
            break;
        case 'open-folder':
            // 调用 IPC handler
            await window.electronAPI.openFolder(contextMenuTargetItem.path);
            break;
        case 'movie-detail':
            // 调用 IPC handler
            await window.electronAPI.openMovieDetail({
                id: contextMenuTargetItem.movieId,
                category: contextMenuTargetItem.category
            });
            break;
        case 'copy-path':
            // 调用 IPC handler
            await window.electronAPI.copyToClipboard(contextMenuTargetItem.path);
            showScreenshotToast('路径已复制到剪贴板');
            break;
    }

    hideContextMenu();
});
```

#### 步骤6: 实现IPC handlers

在 `ipc-handlers.js` 中实现了两个新的IPC处理器：

**open-folder handler**：
```javascript
ipcMain.handle('open-folder', async (event, filePath) => {
    try {
        const { shell } = require('electron');
        const folderPath = path.dirname(filePath);
        await shell.openPath(folderPath);
        return { success: true };
    } catch (error) {
        console.error('Error opening folder:', error);
        return { success: false, error: error.message };
    }
});
```

关键点：
- 使用 `path.dirname(filePath)` 获取文件所在目录
- 使用 `shell.openPath()` 在文件管理器中打开目录
- 返回结果包含成功状态或错误信息

**copy-to-clipboard handler**：
```javascript
ipcMain.handle('copy-to-clipboard', async (event, text) => {
    try {
        const { clipboard } = require('electron');
        clipboard.writeText(text);
        return { success: true };
    } catch (error) {
        console.error('Error copying to clipboard:', error);
        return { success: false, error: error.message };
    }
});
```

关键点：
- 使用 `clipboard.writeText()` 复制文本到剪贴板
- 返回成功状态
